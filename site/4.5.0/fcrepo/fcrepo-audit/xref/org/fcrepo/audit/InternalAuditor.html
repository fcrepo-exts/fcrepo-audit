75      /**
76       * Logger for this class.
77       */
78      private static final Logger LOGGER = getLogger(InternalAuditor.class);
79  
80      private static final String AUDIT_CONTAINER = "fcrepo.audit.container";
81  
82      private static String AUDIT_CONTAINER_LOCATION;
83  
84      @Inject
85      private EventBus eventBus;
86  
87      @Inject
88      private Repository repository;
89  
90      @Inject
91      private ContainerService containerService;
92  
93      private Session session;
94      private static JcrTools jcrTools = new JcrTools(true);
95  
96      /**
97       * Register with the EventBus to receive events.
98       * @throws RepositoryRuntimeException
99       */
100     @PostConstruct
101     public void register() throws RepositoryRuntimeException {
102         try {
103             AUDIT_CONTAINER_LOCATION = System.getProperty(AUDIT_CONTAINER);
104             if (AUDIT_CONTAINER_LOCATION != null) {
105                 LOGGER.info("Initializing: {}", this.getClass().getCanonicalName());
106                 eventBus.register(this);
107                 if (!AUDIT_CONTAINER_LOCATION.startsWith("/")) {
108                     AUDIT_CONTAINER_LOCATION = "/" + AUDIT_CONTAINER_LOCATION;
109                 }
110                 if (AUDIT_CONTAINER_LOCATION.endsWith("/")) {
111                     AUDIT_CONTAINER_LOCATION = AUDIT_CONTAINER_LOCATION.substring(0,
112                             AUDIT_CONTAINER_LOCATION.length() - 2);
113                 }
114                 session = repository.login();
115                 containerService.findOrCreate(session, AUDIT_CONTAINER_LOCATION);
116 
117                 LOGGER.debug("Registering audit CND");
118                 jcrTools.registerNodeTypes(session, "audit.cnd");
119 
120                 session.save();
121             } else {
122                 LOGGER.warn("Cannot Initialize: {}", this.getClass().getCanonicalName());
123                 LOGGER.warn("System property not found: " + AUDIT_CONTAINER);
124             }
125         } catch (RepositoryException e) {
126             throw new RepositoryRuntimeException(e);
127         }
128     }
129 
130     /**
131      * Fedora internal events are received and processed by this method.
132      *
133      * @param event
134      *        The {@link FedoraEvent} to record.
135      * @throws RepositoryRuntimeException on error
136      */
137     @Subscribe
138     public void recordEvent(final FedoraEvent event) throws RepositoryRuntimeException {
139         LOGGER.debug("Event detected: {} {}", event.getUserID(), event.getPath());
140         boolean isParentNodeLastModifiedEvent = false;
141         final String eventType = AuditUtils.getEventURIs(event.getTypes());
142         final Set<String> properties = event.getProperties();
143         if (eventType.contains(AuditProperties.PROPERTY_CHANGED)) {
144             isParentNodeLastModifiedEvent = true;
145             final Iterator<String> propertiesIter = properties.iterator();
146             String property;
147             while (properties.iterator().hasNext()) {
148                 property = propertiesIter.next();
149                 if (!property.equals(AuditProperties.LAST_MODIFIED) &&
150                         !property.equals(AuditProperties.LAST_MODIFIED_BY)) {
151                     /* adding/removing a file updates the lastModified property of the parent container,
152                     so ignore updates when only lastModified is changed */
153                     isParentNodeLastModifiedEvent = false;
154                     break;
155                 }
156             }
157         }
158         if (!event.getPath().startsWith(AUDIT_CONTAINER_LOCATION)
159                 && !isParentNodeLastModifiedEvent) {
160             try {
161                 createAuditNode(event);
162             } catch (IOException e) {
163                 throw new RepositoryRuntimeException(e);
164             }
165         }
166     }
167 
168     /**
169      * Close external connections
170      */
171     @PreDestroy
172     public void releaseConnections() {
173         LOGGER.debug("Tearing down: {}", this.getClass().getCanonicalName());
174         eventBus.unregister(this);
175     }

177     // JCR property name, not URI
178     private static final String PREMIS_OBJ = "premis:hasEventRelatedObject";
179 
180     /**
181      * Creates a node for the audit event under the configured container.
182      *
183      * @param event to be persisted in the repository
184      * @throws RepositoryRuntimeException on error
185      * @throws java.io.IOException on json mapping error
186      */
187     public void createAuditNode(final FedoraEvent event) throws RepositoryRuntimeException, IOException {
188         try {
189             final String userData = event.getUserData();
190             final ObjectMapper mapper = new ObjectMapper();
191             final JsonNode json = mapper.readTree(userData);
192             final String userAgent = json.get("userAgent").asText();
193             String baseURL = json.get("baseURL").asText();
194             if (baseURL.endsWith("/")) {
195                 baseURL = baseURL.substring(0, baseURL.length() - 1);
196             }
197             String path = event.getPath();
198             if ( path.endsWith("/" + JCR_CONTENT) ) {
199                 path = path.replaceAll("/" + JCR_CONTENT,"");
200             }
201             final String uri = baseURL + path;
202             final Long timestamp =  event.getDate();
203             final DateFormat df = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss'Z'");
204             df.setTimeZone(TimeZone.getTimeZone("UTC"));
205             final String eventDate = df.format(new Date(timestamp));
206             final String userID = event.getUserID();
207             final String eventType = AuditUtils.getEventURIs(event.getTypes());
208             final String properties = Joiner.on(',').join(event.getProperties());
209             final String auditEventType = AuditUtils.getAuditEventType(eventType, properties);
210             final FedoraResource auditResource = containerService.findOrCreate(session,
211                     AUDIT_CONTAINER_LOCATION + "/" + event.getEventID());
212 
213             LOGGER.debug("Audit node {} created for event.", event.getEventID());
214 
215             final Model m = createDefaultModel();
216             final String auditResourceURI = baseURL + AUDIT_CONTAINER_LOCATION + "/" + event.getEventID();
217             final Resource s = createResource(auditResourceURI);
218             m.add(createStatement(s, AuditProperties.RDF_TYPE, createResource(AuditProperties.INTERNAL_EVENT)));
219             m.add(createStatement(s, AuditProperties.RDF_TYPE, createResource(AuditProperties.PREMIS_EVENT)));
220             m.add(createStatement(s, AuditProperties.RDF_TYPE, createResource(AuditProperties.PROV_EVENT)));
221             m.add(createStatement(s, AuditProperties.PREMIS_TIME, createTypedLiteral(eventDate, XSDdateTime)));
222             m.add(createStatement(s, AuditProperties.PREMIS_AGENT, createTypedLiteral(userID, XSDstring)));
223             m.add(createStatement(s, AuditProperties.PREMIS_AGENT, createTypedLiteral(userAgent, XSDstring)));
224             if (auditEventType != null) {
225                 m.add(createStatement(s, AuditProperties.PREMIS_TYPE, createResource(auditEventType)));
226             }
227 
228             auditResource.replaceProperties(new PrefixingIdentifierTranslator(session, baseURL + "/"), m,
229                     new RdfStream());
230 
231             // set link to impacted object using a URI property to preserve the link if it's deleted
232             try {
233                 auditResource.setURIProperty(PREMIS_OBJ, new URI(uri));
234             } catch (URISyntaxException e) {
235                 LOGGER.warn("Error creating URI for repository resource {}", uri);
236             }
237 
238             session.save();
239         } catch (RepositoryException e) {
240             throw new RepositoryRuntimeException(e);
241         }
242     }
243 
244     @VisibleForTesting
245     protected Statement createStatement(final Resource subject, final String property, final RDFNode object) {
246         return ResourceFactory.createStatement(subject, createProperty(property), object);
247     }
